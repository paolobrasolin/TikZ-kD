\documentclass{standalone}

\usepackage{tikz,tikz-3dplot}
\usetikzlibrary[kD,calc]
\usetikzlibrary[decorations.markings]
\usetikzlibrary[arrows.meta]

\tikzset{
->-/.style={thick,decoration={
  markings,
  mark=at position .5 with {\arrow{Latex}}},postaction={decorate}},
-<-/.style={thick,decoration={
  markings,
  mark=at position .5 with {\arrow{Latex[reversed]}}},postaction={decorate}},
  ≡>/.style={line width=1pt,double distance=3pt,-Implies,kD/chop=.4-.6},
  dim/.style={dashed},
  ·<·/.style={-<-,dim}
}

\begin{document}
\tdplotsetmaincoords{64}{36}
\begin{tikzpicture}[scale=4,tdplot_main_coords]
  \def\phi{(1+sqrt(5))/2}
  \def\sp{1/(\phi*2)}
  \def\sz{sqrt(\phi/2)}
  \def\hz{sqrt(pow(\phi,3)/2)}
  \def\hp{1/2}
  \foreach \n/\x/\y in {A/-1/-1,B/1/-1,C/-1/1,D/1/1}
    {\coordinate (\n)   at ($0.5*(\x,\y,0)$);
     \coordinate (\n\n) at ($(\n)+\sp*(\x,\y,0)+\sz*(0,0,1)$);}
  \foreach \m/\n/\x/\y in {A/B/0/-1,B/D/1/0,C/D/0/1,A/C/-1/0}
    \coordinate (\m\n) at ($(\m)!0.5!(\n)+\hp*(\x,\y,0)+\hz*(0,0,1)$);
  \foreach \n/\u in {Y/C,X/B}
    \coordinate (\n) at ($(A\u)-(\u\u)+(\u D)$);
  \mor DD ≡>,dim $(C)!.5!(CC)$;                      % back face, right
  \mor AC ≡>,dim $(A)!.5!(C)$;                       % back face, left
  \mor CD -<- DD -<- BD -<- BB;                      % crown, right 
  \mor CC ·<· AC -<- AA -<- AB;                      % crown, left
  \mor C ·<· D -<- B;                                % floor, right
  \mor * ·<· A -<- *;                                % floor, left
  \mor[red]  CD ·<· CC ·<· C ·<· B -<- BB -<- AB;    % source edge
  \mor[blue]  * -<-  Y -<- X -<- *;                  % target edge
  \mor BD -<- X; \mor A -<- AA;
  \mor DD -<- D; \mor Y -<- AC;
  \mor BD ≡> $(B)!.5!(D)$;                           % front face, bottom right
  \mor AA ≡> $(B)!.5!(BB)$;                          % front face, bottom left
  \mor Y  ≡> $(BD)!.5!(DD)$;                         % front face, top right
  \mor X  ≡> $(AA)!.5!(AC)$;                         % front face, top left
\end{tikzpicture}
\end{document}
