\def\rotateRPY#1#2#3%
  {% eat up roll, pitch and yaw
   \pgfmathsetmacro{\R}{#1}
   \pgfmathsetmacro{\P}{#2}
   \pgfmathsetmacro{\Y}{#3}
   % transform x,y and z versors
   \pgfmathsetmacro{\newxx}{cos(\Y)*cos(\P)}
   \pgfmathsetmacro{\newxy}{sin(\Y)*cos(\P)}
   \pgfmathsetmacro{\newxz}{-sin(\P)}
   \pgfmathsetmacro{\newyx}{cos(\Y)*sin(\P)*sin(\R)-sin(\Y)*cos(\R)}
   \pgfmathsetmacro{\newyy}{sin(\Y)*sin(\P)*sin(\R)+cos(\Y)*cos(\R)}
   \pgfmathsetmacro{\newyz}{cos(\P)*sin(\R)}
   \pgfmathsetmacro{\newzx}{cos(\Y)*sin(\P)*cos(\R)+sin(\Y)*sin(\R)}
   \pgfmathsetmacro{\newzy}{sin(\Y)*sin(\P)*cos(\R)-cos(\Y)*sin(\R)}
   \pgfmathsetmacro{\newzz}{cos(\P)*cos(\R)}
   % compute their projections
   \path (\newxx,\newxy,\newxz); \pgfgetlastxy{\nxx}{\nxy};
   \path (\newyx,\newyy,\newyz); \pgfgetlastxy{\nyx}{\nyy};
   \path (\newzx,\newzy,\newzz); \pgfgetlastxy{\nzx}{\nzy};}
